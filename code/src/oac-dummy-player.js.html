(function() {
  var OACDummyPlayerDrv;

  $(document).ready(function() {
    return OAC.Client.StreamingVideo.PlayerController.register("OACDummyPlayerDrv");
  });

  OACDummyPlayerDrv = (function() {
    var OACDummyPlayerDrvBinding;

    function OACDummyPlayerDrv() {
      this.playerObj = null;
      this.oac_version = '1.0';
    }

    OACDummyPlayerDrv.prototype.getAvailablePlayers = function() {
      return $('.dummyplayer');
    };

    OACDummyPlayerDrv.prototype.getOACVersion = function() {
      return this.oac_version;
    };

    OACDummyPlayerDrv.prototype.setDomObj = function(DOMObject) {
      return new OACDummyPlayerDrvBinding($(DOMObject).data('player'));
    };

    OACDummyPlayerDrvBinding = (function() {

      function OACDummyPlayerDrvBinding(player) {
        this.playerObj = player;
      }

      OACDummyPlayerDrvBinding.prototype.onPlayheadUpdate = function(callback) {
        return this.playerObj.onplayheadupdate(callback);
      };

      OACDummyPlayerDrvBinding.prototype.getcoordinates = function() {
        var coordinates;
        coordinates = this.playerObj.getcoordinates();
        coordinates[0] = parseInt(coordinates[0], 10);
        coordinates[1] = parseInt(coordinates[1], 10);
        return coordinates;
      };

      OACDummyPlayerDrvBinding.prototype.getsize = function() {
        var size;
        size = this.playerObj.getsize();
        size[0] = parseInt(size[0], 10);
        size[1] = parseInt(size[1], 10);
        return size;
      };

      OACDummyPlayerDrvBinding.prototype.play = function() {
        return this.playerObj.play();
      };

      OACDummyPlayerDrvBinding.prototype.pause = function() {
        return this.playerObj.pause();
      };

      OACDummyPlayerDrvBinding.prototype.getPlayhead = function() {
        return this.playerObj.getplayhead();
      };

      OACDummyPlayerDrvBinding.prototype.setPlayhead = function(value) {
        return this.playerObj.setplayhead(value);
      };

      return OACDummyPlayerDrvBinding;

    })();

    return OACDummyPlayerDrv;

  })();

}).call(this);
