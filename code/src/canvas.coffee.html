    parseNPT = (npt) ->
      if npt.indexOf(':') == -1
        seconds = parseFloat npt
        minutes = 0
        hours = 0
      else
        bits = ((parseFloat b) for b in npt.split(':'))
        seconds = bits.pop()
        if bits.length > 0
          minutes = bits.pop()
        else
          minutes = 0
        if bits.length > 0
          hours = bits.pop()
        else
          hours = 0
      (hours * 60 + minutes) * 60 + seconds
